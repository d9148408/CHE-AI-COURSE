def preprocess_tep_data(df, handle_missing='drop', remove_outliers=False):
    """
    TEP æ•¸æ“šé è™•ç†
    
    Parameters:
    -----------
    df : pd.DataFrame
        åŸå§‹æ•¸æ“š
    handle_missing : str, default='drop'
        ç¼ºå¤±å€¼è™•ç†æ–¹å¼ ('drop', 'interpolate', 'forward_fill')
    remove_outliers : bool, default=False
        æ˜¯å¦ç§»é™¤æ¥µç«¯ç•°å¸¸å€¼ (ä½¿ç”¨ IQR æ–¹æ³•)
    
    Returns:
    --------
    df_clean : pd.DataFrame
        æ¸…ç†å¾Œçš„æ•¸æ“š
    """
    df_clean = df.copy()
    
    # 1. æª¢æŸ¥ç¼ºå¤±å€¼
    missing_count = df_clean.isnull().sum()
    if missing_count.any():
        print(f"ç™¼ç¾ç¼ºå¤±å€¼: \n{missing_count[missing_count > 0]}")
        
        if handle_missing == 'drop':
            df_clean = df_clean.dropna()
        elif handle_missing == 'interpolate':
            df_clean = df_clean.interpolate(method='linear')
        elif handle_missing == 'forward_fill':
            df_clean = df_clean.fillna(method='ffill')
    
    # 2. ç§»é™¤æ¥µç«¯ç•°å¸¸å€¼ (å¯é¸)
    if remove_outliers:
        feature_cols = [col for col in df_clean.columns 
                       if col.startswith('XMEAS') or col.startswith('XMV')]
        
        for col in feature_cols:
            Q1 = df_clean[col].quantile(0.25)
            Q3 = df_clean[col].quantile(0.75)
            IQR = Q3 - Q1
            lower_bound = Q1 - 3 * IQR
            upper_bound = Q3 + 3 * IQR
            
            # æ¨™è¨˜ç•°å¸¸å€¼ä½†ä¸åˆªé™¤ (ä¿ç•™ç”¨æ–¼ç•°å¸¸æª¢æ¸¬)
            outlier_mask = (df_clean[col] < lower_bound) | (df_clean[col] > upper_bound)
            if outlier_mask.sum() > 0:
                print(f"{col}: {outlier_mask.sum()} outliers detected")
    
    return df_clean
```

def standardize_features(df_train, df_test=None, method='standard'):
    """
    ç‰¹å¾µæ¨™æº–åŒ–
    
    Parameters:
    -----------
    df_train : pd.DataFrame
        è¨“ç·´æ•¸æ“š
    df_test : pd.DataFrame, optional
        æ¸¬è©¦æ•¸æ“š
    method : str, default='standard'
        æ¨™æº–åŒ–æ–¹æ³• ('standard', 'minmax', 'robust')
    
    Returns:
    --------
    df_train_scaled : pd.DataFrame
        æ¨™æº–åŒ–å¾Œçš„è¨“ç·´æ•¸æ“š
    df_test_scaled : pd.DataFrame (if df_test is not None)
        æ¨™æº–åŒ–å¾Œçš„æ¸¬è©¦æ•¸æ“š
    scaler : fitted scaler object
        å·²æ“¬åˆçš„æ¨™æº–åŒ–å™¨
    """
    feature_cols = [col for col in df_train.columns 
                   if col.startswith('XMEAS') or col.startswith('XMV')]
    
    if method == 'standard':
        scaler = StandardScaler()
    elif method == 'minmax':
        scaler = MinMaxScaler()
    elif method == 'robust':
        from sklearn.preprocessing import RobustScaler
        scaler = RobustScaler()
    else:
        raise ValueError(f"Unknown method: {method}")
    
    # æ“¬åˆä¸¦è½‰æ›è¨“ç·´æ•¸æ“š
    df_train_scaled = df_train.copy()
    df_train_scaled[feature_cols] = scaler.fit_transform(df_train[feature_cols])
    
    # è½‰æ›æ¸¬è©¦æ•¸æ“š (ä½¿ç”¨è¨“ç·´æ•¸æ“šçš„çµ±è¨ˆé‡)
    if df_test is not None:
        df_test_scaled = df_test.copy()
        df_test_scaled[feature_cols] = scaler.transform(df_test[feature_cols])
        return df_train_scaled, df_test_scaled, scaler
    
    return df_train_scaled, scaler
```

def evaluate_anomaly_detection(y_true, y_pred, scores, method_name):
    """è©•ä¼°ç•°å¸¸æª¢æ¸¬æ•ˆæœ"""
    from sklearn.metrics import precision_recall_fscore_support
    
    # è¨ˆç®—å„ç¨®æŒ‡æ¨™
    precision, recall, f1, _ = precision_recall_fscore_support(
        y_true, y_pred, average='binary', pos_label=-1
    )
    
    # ROC Curve
    fpr, tpr, _ = roc_curve(y_true, scores, pos_label=-1)
    roc_auc = auc(fpr, tpr)
    
    # Precision-Recall Curve
    precision_curve, recall_curve, _ = precision_recall_curve(y_true, scores, pos_label=-1)
    pr_auc = auc(recall_curve, precision_curve)
    
    metrics = {
        'Method': method_name,
        'Precision': precision,
        'Recall': recall,
        'F1_Score': f1,
        'ROC_AUC': roc_auc,
        'PR_AUC': pr_auc
    }
    
    return metrics, (fpr, tpr), (precision_curve, recall_curve)
